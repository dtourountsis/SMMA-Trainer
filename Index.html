function mulberry32(seed){let t=seed>>>0;return function(){t+=0x6D2B79F5;let r=Math.imul(t^(t>>>15),1|t);r^=r+Math.imul(r^(r>>>7),61|r);return((r^(r>>>14))>>>0)/4294967296;};}

function ri(rng,a,b){return a+Math.floor(rng()*(b-a+1));}

function pk(rng,arr){return arr[Math.floor(rng()*arr.length)];}

function shuf(rng,arr){const a=arr.slice();for(let i=a.length-1;i>0;i--){const j=Math.floor(rng()*(i+1));[a[i],a[j]]=[a[j],a[i]];}return a;}

const ABC="ABCDEFGHIJKLMNOPQRSTUVWXYZ";

const L=i=>ABC[i];

const IX=c=>ABC.indexOf(c);


function finalize(rng,q,answer,rawDistractors){

  const opts=[answer];

  for(const d of rawDistractors){

    if(opts.length>=4)break;

    if(d!==null&&d!==undefined&&d!==answer&&!opts.includes(d))opts.push(d);

  }

  let guard=0;

  while(opts.length<4&&guard++<50){

    const d=q._fallback(rng);

    if(d!==answer&&!opts.includes(d))opts.push(d);

  }

  delete q._fallback;

  q.options=shuf(rng,opts);

  q.answer=answer;

  return q;

}

      _fallback:r=>{const i=ri(r,0,codeB.length-1);const j=Math.min(25,Math.max(0,IX(codeB[i])+pk(r,[1,-1,2])));return codeB.slice(0,i)+L(j)+codeB.slice(i+1);}};

    const d=[wrong(rng),wrong(rng),wrong(rng)];

function genLetterSum(level,rng){

  const val=c=>IX(c)+1;

  for(let attempt=0;attempt<200;attempt++){

    if(level===1){

      const a=L(ri(rng,0,9)),b=L(ri(rng,0,9));

      const ans=val(a)+val(b);

      const q={prompt:"If A = 1, B = 2, C = 3 and so on, what is "+a+" + "+b+"?",mono:a+" + "+b+" = ?",alphabet:true,

        explain:a+" = "+val(a)+" and "+b+" = "+val(b)+", so "+a+" + "+b+" = "+ans+".",

        _fallback:r=>String(ans+ri(r,1,6))};

      return finalize(rng,q,String(ans),[ans+1,ans-1,ans+2].map(String));

    }

    if(level===2){

      if(rng()<0.5){

        const a=L(ri(rng,8,19)),b=L(ri(rng,0,7));

        const op=pk(rng,["+","−"]);

        const ans=op==="+"?val(a)+val(b):val(a)-val(b);

        if(ans<1)continue;

        const q={prompt:"If A = 1, B = 2, C = 3 and so on, what is "+a+" "+op+" "+b+"?",mono:a+" "+op+" "+b+" = ?",alphabet:true,

          explain:a+" = "+val(a)+" and "+b+" = "+val(b)+", so the answer is "+ans+".",

          _fallback:r=>String(Math.max(1,ans+ri(r,1,5)))};

        return finalize(rng,q,String(ans),[ans+1,ans-1,ans+2].map(String));

      }

      const a=L(ri(rng,0,11)),b=L(ri(rng,0,11));

      const sum=val(a)+val(b);

      if(sum>26)continue;

      const ans=L(sum-1);

      const q={prompt:"If A = 1, B = 2, C = 3 and so on, which letter has the same value as "+a+" + "+b+"?",mono:a+" + "+b+" = ?",alphabet:true,

        explain:a+" ("+val(a)+") + "+b+" ("+val(b)+") = "+sum+", and the letter with value "+sum+" is "+ans+".",

        _fallback:r=>L(ri(r,0,25))};

      const d=[sum,sum-2,sum+1].filter(x=>x>=1&&x<=26).map(x=>L(x-1));

      return finalize(rng,q,ans,d);

    }

    const a=L(ri(rng,2,12)),b=L(ri(rng,2,12)),c=L(ri(rng,0,9));

    const res=val(a)+val(b)-val(c);

    if(res<1||res>26)continue;

    const ans=L(res-1);

    const q={prompt:"If A = 1, B = 2, C = 3 and so on, which letter has the same value as "+a+" + "+b+" − "+c+"?",mono:a+" + "+b+" − "+c+" = ?",alphabet:true,

      explain:val(a)+" + "+val(b)+" − "+val(c)+" = "+res+", and the letter with value "+res+" is "+ans+".",

      _fallback:r=>L(ri(r,0,25))};

    const d=[res,res-2,res+1].filter(x=>x>=1&&x<=26).map(x=>L(x-1));

    return finalize(rng,q,ans,d);

  }

  throw new Error("letterSum generation failed");

}

const RN_RULES=[

];

function genRelatedNumbers(level,rng){

  const rules=RN_RULES.filter(r=>r.lv===level);

  for(let attempt=0;attempt<400;attempt++){

    const R=pk(rng,rules);

    const mk=()=>{for(let t=0;t<60;t++){const a=ri(rng,2,9),b=ri(rng,2,12);if(R.ok(a,b)&&R.f(a,b)>=1&&R.f(a,b)<=99&&Number.isInteger(R.f(a,b)))return[a,b];}return null;};

    const p1=mk(),p2=mk(),pq=mk();

    if(!p1||!p2||!pq)continue;

    if((p1[0]===p2[0]&&p1[1]===p2[1])||(p1[0]===pq[0]&&p1[1]===pq[1])||(p2[0]===pq[0]&&p2[1]===pq[1]))continue;

    const m1=R.f(...p1),m2=R.f(...p2),ans=R.f(...pq);

    let collision=false;

    for(const R2 of RN_RULES){

      if(R2.id===R.id)continue;

      if(R2.ok(...p1)&&R2.ok(...p2)&&R2.f(...p1)===m1&&R2.f(...p2)===m2&&Number.isInteger(R2.f(...p1))){

        if(!R2.ok(...pq)||R2.f(...pq)!==ans){collision=true;break;}

      }

    }

    if(collision)continue;

    const tri=(a,m,b)=>"( "+a+"  [ "+m+" ]  "+b+" )";

    const q={prompt:"The three numbers in each group are related in the same way. What number replaces the question mark?",

      mono:tri(p1[0],m1,p1[1])+"   "+tri(p2[0],m2,p2[1])+"   "+tri(pq[0],"?",pq[1]),alphabet:false,

      explain:"In each group, the middle number is "+R.text+": "+R.eg(p1[0],p1[1],m1)+" and "+R.eg(p2[0],p2[1],m2)+". So here it is "+R.eg(pq[0],pq[1],ans)+".",

      _fallback:r=>String(ans+ri(r,1,9))};

    const d=[];

    for(const R2 of RN_RULES){if(R2.id!==R.id&&R2.ok(...pq)){const v=R2.f(...pq);if(Number.isInteger(v)&&v>=1&&v<=99)d.push(String(v));}}

    d.push(String(ans+1),String(ans-1));

    return finalize(rng,q,String(ans),shuf(rng,d.filter(x=>Number(x)>=0)));

  }

  throw new Error("relatedNumbers generation failed");

}

function formatCurated(t, item, rng) {

  const base = { typeId: t, level: item.level, curatedId: item.id, explain: item.explain, answer: item.answer, options: shuf(rng, item.options) };

  switch (t) {

    case "synonym":      return { ...base, prompt: "Which word is closest in meaning to the word in capitals?", big: item.target };

    case "antonym":      return { ...base, prompt: "Which word is most opposite in meaning to the word in capitals?", big: item.target };

    case "oddOneOut":    return { ...base, prompt: "Four of these words go together. Which one does not belong?" };

    case "analogy":      return { ...base, prompt: "Complete the sentence in the most sensible way.", line: item.a + " is to " + item.b + "  as  " + item.c + " is to …" };

    case "compound":     return { ...base, prompt: "Which word joins on to the word in capitals to make one new, real word?", big: item.stem + " + ___" };

    case "hiddenWord":   return { ...base, prompt: "A four-letter word is hidden at the end of one word and the start of the next. Which is it?", line: "“" + item.sentence + "”" };

    case "moveLetter":   return { ...base, prompt: "Move one letter from the first word into the second word, so that both make new words. Which letter moves?", mono: item.from + "   →   " + item.to };

    case "insertLetter": return { ...base, prompt: "The same letter finishes the first word and starts the second word — in both pairs. Which letter is it?", mono: item.pairs.map(p => p[0] + " _ " + p[1]).join("      ") };

    case "missingThree": return { ...base, prompt: "Three letters in a row have been removed from the word in capitals. The missing letters spell a real three-letter word. Which?", line: "“" + item.sentence + "”" };

    case "doubleMeaning":return { ...base, prompt: "Which word goes equally well with both pairs of words?", line: "(" + item.pairA.join(", ") + ")      (" + item.pairB.join(", ") + ")" };

    default:             return base;

  }

}

function pickCurated(t, level, rng, st) {

  const bank = BANKS[t];

  const rec = st.recent[t] || [];

  let pool = bank.filter(i => i.level === level);

  if (!pool.length) pool = bank.filter(i => Math.abs(i.level - level) === 1);

  if (!pool.length) pool = bank;

  let fresh = pool.filter(i => !rec.includes(i.id));

  if (!fresh.length) fresh = pool;

  const item = pk(rng, fresh);

  st.recent[t] = [item.id, ...rec].slice(0, 10);

  return formatCurated(t, item, rng);

}

function makeQuestion(t, level, rng, st) {

  if (GENERATORS[t]) { const q = GENERATORS[t](level, rng); q.typeId = t; q.level = level; return q; }

  return pickCurated(t, level, rng, st);

}

function mergeState(loaded) {

  const d = defaultState();

}

function recordAnswer(st, t, ok) {

  st.totals.answered += 1; if (ok) st.totals.correct += 1;

  const s = st.types[t];

  s.answered += 1; if (ok) s.correct += 1;

  s.win = [...s.win, ok ? 1 : 0].slice(-8);

  if (s.win.length === 8) {

    const sum = s.win.reduce((a, b) => a + b, 0);

    if (sum >= 7 && s.level < 3) { s.level += 1; s.win = []; }

    else if (sum <= 3 && s.level > 1) { s.level -= 1; s.win = []; }

  }

}

function typeAccuracy(st, t) { const s = st.types[t]; return (s.correct + 3) / (s.answered + 5); }

function pickWeighted(rng, entries) {

  const tot = entries.reduce((a, e) => a + e[1], 0); let r = rng() * tot;

  for (const [t, w] of entries) { r -= w; if (r <= 0) return t; }

  return entries[entries.length - 1][0];

}

function buildDrill(st, rng) {

  const counts = {}; const qs = [];

  while (qs.length < 12) {

    const entries = ALL_TYPES.filter(t => (counts[t] || 0) < 2).map(t => [t, Math.max(0.05, 1.06 - typeAccuracy(st, t))]);

    const t = pickWeighted(rng, entries);

    counts[t] = (counts[t] || 0) + 1;

    qs.push(makeQuestion(t, st.types[t].level, rng, st));

  }

  return shuf(rng, qs);

}

function buildFocus(st, rng, t) {

  const qs = []; for (let i = 0; i < 8; i++) qs.push(makeQuestion(t, st.types[t].level, rng, st));

  return qs;

}

function buildMock(st, rng, kind) {

  const qs = list.map(t => makeQuestion(t, rng() < 0.7 ? 3 : 2, rng, st));

  return shuf(rng, qs);

}

/* ================= PRACTICE FLOW (drill + focus) ================= */

function PracticeFlow({ st, commit, kind, focusType, onDone, onExit }) {

  const [qs, setQs] = useState(null);

  const [idx, setIdx] = useState(0);

  const [picked, setPicked] = useState(null);

  const [score, setScore] = useState(0);

  useEffect(() => {

    const next = structuredClone(st);

    const rng = makeRng();

    const built = kind === "drill" ? buildDrill(next, rng) : buildFocus(next, rng, focusType);

    commit(next);

    setQs(built);

  }, []);

  if (!qs) return null;

  const q = qs[idx];

  const locked = picked !== null;


  function handlePick(opt) {

    if (locked) return;

    setPicked(opt);

    const ok = opt === q.answer;

    if (ok) setScore(s => s + 1);

    const next = structuredClone(st);

    recordAnswer(next, q.typeId, ok);

    commit(next);

  }

  function handleNext() {

    if (idx + 1 < qs.length) { setIdx(idx + 1); setPicked(null); return; }

    const next = structuredClone(st);

    if (kind === "drill") {

      next.totals.drills += 1;

      const today = todayStr();

      if (next.streak.last !== today) {

        next.streak.current = next.streak.last === yesterdayStr() ? next.streak.current + 1 : 1;

        next.streak.best = Math.max(next.streak.best, next.streak.current);

        next.streak.last = today;

      }

    }

    commit(next);

    onDone({ score, total: qs.length });

  }


  return (

    <div>

      <TopBar

        left={<ExitButton onExit={onExit} />}

        centre={<span className="font-bold text-[15px]" style={{ color: "var(--muted)" }}>{idx + 1} of {qs.length}</span>}

        right={<span className="vr-chip">{score} right</span>}

      />

      <div className="vr-bar mb-4"><div style={{ width: ((idx + (locked ? 1 : 0)) / qs.length * 100) + "%" }} /></div>

      <QuestionCard q={q} picked={picked} onPick={handlePick} locked={locked} showFeedback={true} />

      {locked && (

        <div className="flex justify-end mt-4">

          <button className="vr-btn vr-btn-primary" onClick={handleNext}>{idx + 1 < qs.length ? "Next" : "Finish"}<ArrowRight size={18} /></button>

      )}

    </div>

  );

}

  const deadlineRef = useRef(null);

  const [idx, setIdx] = useState(0);

  const [answers, setAnswers] = useState({});

  const answersRef = useRef({});

  const qsRef = useRef(null);

  const submittedRef = useRef(false);

    const next = structuredClone(st);

    const built = buildMock(next, makeRng(), kind);

    commit(next);

    qsRef.current = built;

    setQs(built);

    deadlineRef.current = Date.now() + totalSecs * 1000;

  }, []);


  function submit() {

    if (submittedRef.current || !qsRef.current) return;

    submittedRef.current = true;

    const finalAnswers = answersRef.current;

    let score = 0; const perType = {};

    qsRef.current.forEach((q, i) => {

      const ok = finalAnswers[i] === q.answer;

      if (ok) score += 1;

function App() {

  const [st, setSt] = useState(null);

  const [memOnly, setMemOnly] = useState(false);

  const [screen, setScreen] = useState({ name: "home" });

  const [sessionKey, setSessionKey] = useState(0);


  useEffect(() => { loadState().then(({ st, ok }) => { setSt(st); setMemOnly(!ok); }); }, []);


  function commit(next) { setSt(next); saveState(next); }

  function go(name, payload) { setSessionKey(k => k + 1); setScreen({ name, payload }); }


  if (!st) return (

    <div className="vr-root flex items-center justify-center" style={{ minHeight: "100vh" }}>

      <style>{STYLE}</style>

      <div className="vr-display" style={{ fontSize: 22, color: "var(--muted)" }}>Sharpening pencils…</div>

    </div>

  );


  const s = screen.name;

  return (

    <div className="vr-root">

      <style>{STYLE}</style>

      <div className="max-w-2xl mx-auto px-4 pb-16">

        {s === "home" && <HomeScreen st={st} go={go} memOnly={memOnly} />}

        {s === "drill" && <PracticeFlow key={sessionKey} st={st} commit={commit} kind="drill" onExit={() => go("home")} onDone={r => go("done", { r, kind: "drill" })} />}

        {s === "focusPick" && <FocusPicker st={st} onPick={t => go("focus", { t })} onBack={() => go("home")} />}

        {s === "focus" && <PracticeFlow key={sessionKey} st={st} commit={commit} kind="focus" focusType={screen.payload.t} onExit={() => go("focusPick")} onDone={r => go("done", { r, kind: "focus", t: screen.payload.t })} />}

        {s === "done" && <SessionDone result={screen.payload.r} kind={screen.payload.kind} streak={st.streak.current} onHome={() => go("home")} onAgain={() => go(screen.payload.kind === "drill" ? "drill" : "focus", screen.payload.kind === "drill" ? undefined : { t: screen.payload.t })} />}

        {s === "mockSetup" && <MockSetup onStart={kind => go("mock", { kind })} onBack={() => go("home")} />}

        {s === "mock" && <MockFlow key={sessionKey} st={st} commit={commit} kind={screen.payload.kind} onExit={() => go("home")} onFinish={data => go("mockReview", data)} />}

        {s === "mockReview" && <MockReview data={screen.payload} onHome={() => go("home")} />}

        {s === "progress" && <ProgressDash st={st} onBack={() => go("home")} memOnly={memOnly} />}

        {s === "about" && <AboutTest onBack={() => go("home")} />}

      </div>

    </div>

  );

}


createRoot(document.getElementById("root")).render(<App />);
